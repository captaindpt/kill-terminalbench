def gen_check_pattern(king_file, king_rank_idx, attacker_pos_in_board, attacker_pattern, 
                      path_positions=None):
    """
    Generate a regex that matches a FEN line where the white king at (king_file, king_rank_idx)
    is attacked by a piece matching attacker_pattern at attacker_pos_in_board.
    path_positions: list of positions that must be empty (for sliders)
    
    If matched, prepend "#" to mark as illegal.
    """
    king_pos = pos_in_board(king_file, king_rank_idx)
    
    match_squares = [(king_pos, 'K'), (attacker_pos_in_board, attacker_pattern)]
    if path_positions:
        for pp in path_positions:
            match_squares.append((pp, r'\.'))
    
    # Deduplicate
    seen = {}
    for pos, pat in match_squares:
        if pos in seen:
            if seen[pos] != pat:
                return  # conflict
        else:
            seen[pos] = pat
    match_squares = list(seen.items())
    match_squares.sort(key=lambda x: x[0])
    
    board_pattern, group_infos, num_groups = make_move_regex(match_squares)
    
    # Match the full line (no newlines)
    # The line is: BOARD b CASTLING EP 0 0
    gn = num_groups
    rest_pat = '( b [KQkq]{1,4}|- [a-h][ZY]|- 0 0)'
    # Actually let me be less strict about the rest
    rest_pat = '([^\n]*)'
    gn += 1
    rest_gn = gn
    
    full_pat = board_pattern + rest_pat
    
    # Replacement: prepend "#" to mark as illegal
    # We need to reconstruct the line with "#" at the start
    board_repl = make_replacement(group_infos, {}, num_groups)  # no changes to board
    full_repl = '#' + board_repl + '\\' + str(rest_gn)
    
    add(full_pat, full_repl)
