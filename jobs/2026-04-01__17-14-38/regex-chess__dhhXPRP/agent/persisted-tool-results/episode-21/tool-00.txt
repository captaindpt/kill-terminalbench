rules = []

def add(pattern, repl):
    rules.append([pattern, repl])

BOARD_OR_SLASH = r'[rnbqkpRNBQKP./]'

def exact_gap(n):
    if n == 0:
        return ''
    if n == 1:
        return BOARD_OR_SLASH
    return BOARD_OR_SLASH + '{' + str(n) + '}'

def pos_in_board(file, rank_idx):
    return rank_idx * 9 + file

def chess_rank(rank_idx):
    return 8 - rank_idx

BLACK_PIECE = r'[rnbqkp]'
EMPTY = r'\.'
BLACK_OR_EMPTY = r'[rnbqkp.]'

def make_move_regex(squares_and_patterns, board_len=71):
    squares_and_patterns = sorted(set(squares_and_patterns), key=lambda x: x[0])
    parts = []
    group_infos = []
    group_num = 0
    prev_end = 0
    
    for pos, pat in squares_and_patterns:
        gap = pos - prev_end
        if gap > 0:
            parts.append('(' + exact_gap(gap) + ')')
            group_num += 1
            group_infos.append(('gap', group_num, prev_end))
        parts.append('(' + pat + ')')
        group_num += 1
        group_infos.append(('sq', group_num, pos))
        prev_end = pos + 1
    
    remaining = board_len - prev_end
    if remaining > 0:
        parts.append('(' + exact_gap(remaining) + ')')
        group_num += 1
        group_infos.append(('gap', group_num, prev_end))
    
    return ''.join(parts), group_infos, group_num

def make_replacement(group_infos, replacements, total_groups):
    parts = []
    for info_type, gnum, pos in group_infos:
        if info_type == 'sq' and pos in replacements:
            parts.append(replacements[pos])
        else:
            parts.append('\\' + str(gnum))
    return ''.join(parts)

def gen_move_rule(src_file, src_rank_idx, dst_file, dst_rank_idx, 
                  piece, capture_pattern, 
                  path_squares=None,
                  extra_squares=None,
                  ep_cond=None,
                  new_ep=None,
                  promotion=None,
                  src_becomes='.',
                  dst_becomes=None):
    
    if dst_becomes is None:
        dst_becomes = promotion if promotion else piece
    
    src_pos = pos_in_board(src_file, src_rank_idx)
    dst_pos = pos_in_board(dst_file, dst_rank_idx)
    
    match_squares = []
    match_squares.append((src_pos, re.escape(piece)))
    match_squares.append((dst_pos, capture_pattern))
    
    replacements = {src_pos: src_becomes, dst_pos: dst_becomes}
    
    if path_squares:
        for pf, pr in path_squares:
            pp = pos_in_board(pf, pr)
            match_squares.append((pp, r'\.'))
            replacements[pp] = '.'
    
    if extra_squares:
        for pp, match, repl in extra_squares:
            match_squares.append((pp, match))
            replacements[pp] = repl
    
    # Deduplicate - if same position appears multiple times, keep the one with the most specific pattern
    seen = {}
    for pos, pat in match_squares:
        if pos not in seen:
            seen[pos] = pat
        else:
            # If same pos has different patterns, we have a conflict - skip this move as invalid
            if seen[pos] != pat:
                return  # conflicting requirements
    match_squares = list(seen.items())
    match_squares.sort(key=lambda x: x[0])
    
    board_pattern, group_infos, num_groups = make_move_regex(match_squares)
    board_replacement = make_replacement(group_infos, replacements, num_groups)
    
    gn = num_groups
    
    # Castling field - just capture it
    castling_pat = '([KQkq]{1,4}|-)'
    gn += 1
    castling_group = gn
    
    # EP field
    if ep_cond:
        ep_pat = '(' + ep_cond + ')'
    else:
        ep_pat = '([a-h][ZY]|-)'
    gn += 1
    ep_group = gn
    
    full_pattern = '!' + board_pattern + r' w ' + castling_pat + ' ' + ep_pat + ' 0 0'
    
    castling_repl = '\\' + str(castling_group)
    ep_repl = new_ep if new_ep else '-'
    
    new_fen = board_replacement + ' b ' + castling_repl + ' ' + ep_repl + ' 0 0'
    full_replacement = '\\g<0>\\n' + new_fen
    
    add(full_pattern, full_replacement)

def castling_removal_rule(board_pos, expected_char, castling_chars_to_check):
    """
    For each generated FEN line: if the board character at board_pos is NOT expected_char,
    remove castling_chars from the castling field.
    
    castling_chars_to_check: string of chars to remove (e.g., "KQ")
    """
    # Match the line with the board split at board_pos
    # Board: [0..board_pos-1] [board_pos] [board_pos+1..70]
    # Then: " b " castling " " ep " 0 0"
    
    before = board_pos
    after = 70 - board_pos
    
    for ch in castling_chars_to_check:
        # Pattern: match a line where:
        # - board[pos] != expected_char
        # - castling field contains ch
        # Replace: remove ch from castling field
        
        # We need to handle the castling field. It can be 1-4 chars from {K,Q,k,q} or "-".
        # We need to remove a specific char.
        
        # Strategy: match the castling field and capture parts before and after ch.
        # Castling field pattern with ch: ([KQkq]*)ch([KQkq]*)
        # This assumes ch appears exactly once (which it does in valid FEN).
        
        # Full pattern:
        # (BOARD_before)(NOT_expected)(BOARD_after) b (CAST_before)ch(CAST_after) (EP) 0 0
        
        not_expected = '[^' + expected_char + r'\n]'
        
        # Groups: 1=board_before, 2=bad_char, 3=board_after, 4=cast_before, 5=cast_after, 6=ep
        pat = (
            '(' + exact_gap(before) + ')' +  # group 1
            '(' + not_expected + ')' +  # group 2
            '(' + exact_gap(after) + ')' +  # group 3
            ' b ' +
            '([KQkq]*)' + re.escape(ch) + '([KQkq]*)' +  # groups 4, 5
            ' ([a-h][ZY]|-) 0 0'  # group 6
        )
        
        repl = '\\1\\2\\3 b \\4\\5 \\6 0 0'
        add(pat, repl)

                      path_positions=None):

    # Deduplicate
    seen = {}
    for pos, pat in match_squares:
        if pos in seen:
            if seen[pos] != pat:
                return  # conflict
        else:
            seen[pos] = pat
    match_squares = list(seen.items())
    match_squares.sort(key=lambda x: x[0])
    
    board_pattern, group_infos, num_groups = make_move_regex(match_squares)
    
    # Match the full line (no newlines)
    # The line is: BOARD b CASTLING EP 0 0
    gn = num_groups
    rest_pat = '( b [KQkq]{1,4}|- [a-h][ZY]|- 0 0)'
    # Actually let me be less strict about the rest
    rest_pat = '([^\n]*)'
    gn += 1
    rest_gn = gn
    
    full_pat = board_pattern + rest_pat
    
    # Replacement: prepend "#" to mark as illegal
    # We need to reconstruct the line with "#" at the start
    board_repl = make_replacement(group_infos, {}, num_groups)  # no changes to board
    full_repl = '#' + board_repl + '\\' + str(rest_gn)
    
    add(full_pat, full_repl)

        for df in [-1, 1]:
